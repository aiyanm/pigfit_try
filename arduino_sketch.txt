
// --- BLE SERVER CODE FOR PIGFIT WEARABLE ---
// Project: PigFit Thesis - Real-time monitoring of livestock health and behavior.
// GOAL: Send raw sensor data via BLE using efficient binary protocol.

// ----------------------------------------------------
// 1. LIBRARIES & CONFIGURATION
// ----------------------------------------------------
#define MAXIMWIRE_EXTERNAL_PULLUP

#include "Arduino_BMI270_BMM150.h"
#include <Arduino_HS300x.h>
#include <MaximWire.h>
#include <ArduinoBLE.h>

// --- One-Wire Setup ---
#define PIN_BUS 2
MaximWire::Bus bus(PIN_BUS);
MaximWire::DS18B20 device;

// --- BLE CONSTANTS ---
const char *BLE_DEVICE_NAME = "PigFit_Device";
const char *BLE_SERVICE_UUID = "6E400001-B5A3-F393-E0A9-E50E24DCCA9E";
const char *BLE_CHARACTERISTIC_UUID = "6E400003-B5A3-F393-E0A9-E50E24DCCA9E";

// Binary packet constants
#define PACKET_SIZE 48
#define MAGIC_NUMBER 0xAA
#define PACKET_VERSION 0x02

// Create BLE Service and Characteristic
BLEService pigFitService(BLE_SERVICE_UUID);
BLECharacteristic dataCharacteristic(BLE_CHARACTERISTIC_UUID, BLERead | BLENotify, PACKET_SIZE);

// --- SENSOR STATE VARIABLES ---
float ds18b20_temp = 0.0f;
float hs300x_temp = 0.0f;
float humidity = 0.0f;
float imu_accel_x = 0.0f;
float imu_accel_y = 0.0f;
float imu_accel_z = 0.0f;
float imu_gyro_x = 0.0f;
float imu_gyro_y = 0.0f;
float imu_gyro_z = 0.0f;
float activityIntensity = 0.0f; // Calculated from acceleration magnitude
float pitchAngle = 0.0f;        // Calculated from accelerometer
float feed_amount = 2.1f;

// Binary packet buffer
uint8_t packet[PACKET_SIZE];

// ----------------------------------------------------
// 2. SETUP & INITIALIZATION
// ----------------------------------------------------

void setup()
{
  Serial.begin(9600);
  // Wait max 2 seconds for Serial - allows battery-only operation
  unsigned long start = millis();
  while (!Serial && millis() - start < 2000)
    ;
  Serial.println("=== PigFit BLE Monitor (BINARY MODE) ===");

  initializeSensors();
  setupBLE();
}

void initializeSensors()
{
  if (!HS300x.begin())
  {
    Serial.println("FATAL: HS300x failed!");
  }
  else
  {
    Serial.println("✓ HS300x OK");
  }

  if (!IMU.begin())
  {
    Serial.println("FATAL: IMU failed!");
  }
  else
  {
    Serial.println("✓ IMU OK");
  }

  if (bus.Discover().FindNextDevice(device) && device.GetModelCode() == MaximWire::DS18B20::MODEL_CODE)
  {
    Serial.println("✓ DS18B20 OK");
    device.Update(bus); // Initial blocking conversion on startup only
  }
  else
  {
    Serial.println("⚠ DS18B20 not found");
    device.Reset();
  }
}

void setupBLE()
{
  if (!BLE.begin())
  {
    Serial.println("FATAL: BLE failed!");
    while (1)
      ;
  }

  BLE.setLocalName(BLE_DEVICE_NAME);
  BLE.setAdvertisedService(pigFitService);
  pigFitService.addCharacteristic(dataCharacteristic);
  BLE.addService(pigFitService);
  BLE.advertise();

  Serial.println("✓ BLE Advertising");
  Serial.print("Device: ");
  Serial.println(BLE_DEVICE_NAME);
}

// ----------------------------------------------------
// 3. MAIN LOOP
// ----------------------------------------------------

void loop()
{
  BLEDevice central = BLE.central();

  if (central)
  {
    Serial.print("Connected: ");
    Serial.println(central.address());
    Serial.println(">>> Entering data transmission loop");

    while (central.connected())
    {
      readSensors();
      packageDataToBinary();

      Serial.print("Sending ");
      Serial.print(PACKET_SIZE);
      Serial.println(" bytes...");

      dataCharacteristic.writeValue(packet, PACKET_SIZE);

      Serial.println("✓ Binary packet sent");

      // Start next DS18B20 conversion NOW so it completes during the delay (~750ms)
      if (device.IsValid())
      {
        device.Update(bus);
      }

      delay(250); // Reduced: conversion takes ~750ms, total cycle ~1s
    }

    Serial.println("Disconnected");

    // Re-start advertising so the app can find it again
    BLE.advertise();
    Serial.println("✓ BLE Advertising (waiting for connection)");
  }

  delay(200);
}

// ----------------------------------------------------
// 4. SENSOR READING
// ----------------------------------------------------

void readSensors()
{
  // Read HS300x
  humidity = HS300x.readHumidity();
  hs300x_temp = HS300x.readTemperature();

  // Read DS18B20 - reads result of PREVIOUS conversion (non-blocking)
  if (device.IsValid())
  {
    float temp = device.GetTemperature<float>(bus);
    if (!isnan(temp))
    {
      ds18b20_temp = temp;
    }
    else
    {
      ds18b20_temp = 0.0f;
    }
  }
  else
  {
    if (bus.Discover().FindNextDevice(device) && device.GetModelCode() == MaximWire::DS18B20::MODEL_CODE)
    {
      device.Update(bus); // blocking only on reconnect
    }
  }

  // Read IMU
  if (IMU.accelerationAvailable())
  {
    IMU.readAcceleration(imu_accel_x, imu_accel_y, imu_accel_z);

    // Calculate Activity Intensity (magnitude of acceleration vector)
    // Formula: sqrt(x^2 + y^2 + z^2)
    activityIntensity = sqrt(imu_accel_x * imu_accel_x +
                             imu_accel_y * imu_accel_y +
                             imu_accel_z * imu_accel_z);

    // Calculate Pitch Angle (tilt angle in degrees)
    // Formula: atan2(Z, sqrt(X^2 + Y^2)) * 180/PI
    float xy_magnitude = sqrt(imu_accel_x * imu_accel_x + imu_accel_y * imu_accel_y);
    pitchAngle = atan2(imu_accel_z, xy_magnitude) * 180.0 / PI;
  }
  else
  {
    imu_accel_x = imu_accel_y = imu_accel_z = 0.0f;
    activityIntensity = 0.0f;
    pitchAngle = 0.0f;
  }

  if (IMU.gyroscopeAvailable())
  {
    IMU.readGyroscope(imu_gyro_x, imu_gyro_y, imu_gyro_z);
  }
  else
  {
    imu_gyro_x = imu_gyro_y = imu_gyro_z = 0.0f;
  }
}

// ----------------------------------------------------
// 5. BINARY PACKAGING
// ----------------------------------------------------

// CRC16 calculation (CCITT)
uint16_t calculateCRC16(const uint8_t *data, uint8_t length)
{
  uint16_t crc = 0xFFFF;

  for (uint8_t i = 0; i < length; i++)
  {
    crc ^= (uint16_t)data[i] << 8;
    for (uint8_t j = 0; j < 8; j++)
    {
      if (crc & 0x8000)
      {
        crc = (crc << 1) ^ 0x1021;
      }
      else
      {
        crc <<= 1;
      }
    }
  }

  return crc;
}

// Copy float to buffer (little-endian)
void writeFloat(uint8_t *buffer, uint8_t offset, float value)
{
  memcpy(buffer + offset, &value, sizeof(float));
}

// Copy uint16 to buffer (little-endian)
void writeUint16(uint8_t *buffer, uint8_t offset, uint16_t value)
{
  buffer[offset] = value & 0xFF;
  buffer[offset + 1] = (value >> 8) & 0xFF;
}

void packageDataToBinary()
{
  // Header
  packet[0] = MAGIC_NUMBER;
  packet[1] = PACKET_VERSION;

  // Data payload
  writeFloat(packet, 2, ds18b20_temp);   // Body temperature (bytes 2-5)
  writeFloat(packet, 6, hs300x_temp);    // Environmental temperature (bytes 6-9)
  writeFloat(packet, 10, humidity);      // Humidity (bytes 10-13)
  writeFloat(packet, 14, imu_accel_x);   // Accelerometer X (bytes 14-17)
  writeFloat(packet, 18, imu_accel_y);   // Accelerometer Y (bytes 18-21)
  writeFloat(packet, 22, imu_accel_z);   // Accelerometer Z (bytes 22-25)
  writeFloat(packet, 26, imu_gyro_x);    // Gyroscope X (bytes 26-29)
  writeFloat(packet, 30, imu_gyro_y);    // Gyroscope Y (bytes 30-33)
  writeFloat(packet, 34, imu_gyro_z);    // Gyroscope Z (bytes 34-37)
  writeFloat(packet, 38, pitchAngle);    // Pitch angle in degrees (bytes 38-41)

  // Compact boolean flag for feeding posture
  packet[42] = feed_amount > 0.1f ? 1 : 0;

  // Padding (bytes 43-45 unused)
  packet[43] = 0;
  packet[44] = 0;
  packet[45] = 0;

  // Calculate and append CRC16
  uint16_t crc = calculateCRC16(packet, 46);
  writeUint16(packet, 46, crc);
}
